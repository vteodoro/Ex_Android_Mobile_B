MainActivity:

    private MenuItem criar, salvar, deletar;
    private List<Long> listaDeletados= new ArrayList<>();


        listView.setOnItemLongClickListener(new AdapterView.OnItemLongClickListener() {
            @Override
            public boolean onItemLongClick(AdapterView<?> parent, View view, int position, long id) {

                criar.setVisible(false);
                salvar.setVisible(false);
                deletar.setVisible(true);

                if(listaDeletados.contains(itemLista.getItemId(position))){
                    deletar.setChecked(false);
                    listaDeletados.remove(itemLista.getItemId(position));
                }else{
                    deletar.setChecked(true);
                    listaDeletados.add(itemLista.getItemId(position));
                }
                if(listaDeletados.isEmpty()){
                    recreate();
                }
                return true;
            }
        });
